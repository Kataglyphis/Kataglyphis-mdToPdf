% Pandoc renders fenced code blocks as:
%   \begin{Shaded}\begin{Highlighting}...\end{Highlighting}\end{Shaded}
% The --highlight-style file defines Shaded via framed/snugshade by default.
% Override Shaded to a tcolorbox with rounded corners.
\AtBeginDocument{%
    \renewenvironment{Shaded}%
    {\begin{tcolorbox}[
        enhanced,
        breakable,
        boxrule=1.3pt,
        colback=shadecolor,
        colframe=myGreenAccent,
        arc=2mm,
        left=3.0mm,
        right=3.0mm,
        top=3.0mm,
        bottom=3.0mm
    ]}%
    {\end{tcolorbox}}%
}
